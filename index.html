print('')
print("BEGINNING OF K-TRUSS ALGORITHM")
meet_the_condition = True
k_value = 3
while (k_value != 5):
    print('')
    print("K-truss with k value = " + str(k_value))

    inc_matrix_co = inc_matrix.copy()
    inc_matrix_co_sparse = csr_matrix(inc_matrix_co)
    # R = EA
    R = inc_matrix_co_sparse.dot(adj_matrix)
    # Replace the value of 2 by 1 and others by 0
    s = R.copy()
    s[s < 2] = 0
    s[s == 2] = 1
    # Initializing a matrix of 1
    I = np.ones((s.shape[1], 1))
    I = csr_matrix(I)
    s = csr_matrix(s)
    x = s.dot(I)
    x = x.todense()

    while (np.min(x) < k_value - 2):
        i_x = np.where(x >= k_value - 2)[0]
        i_xc = np.where(x < k_value - 2)[0]
        inc_matrix_x = np.delete(inc_matrix_co, 
                                i_x, 
                                axis = 0)
        inc_matrix_co = np.delete(inc_matrix_co, 
                            i_xc, 
                            axis = 0)
        adj_matrix_x = np.zeros((inc_matrix_x.shape[1], 
                                inc_matrix_x.shape[1]),
                                dtype = int)
        
        for i in range(inc_matrix_x.shape[0]):
            position = np.asarray(np.nonzero(inc_matrix_x[i]))
            adj_matrix_x[position[0][0]][position[0][1]] = 1
            adj_matrix_x[position[0][1]][position[0][0]] = 1

        R = np.delete(R, i_xc, axis = 0) #Rxc           
        inc_matrix_co_sparse = csr_matrix(inc_matrix_co) 
        R = R - inc_matrix_co_sparse.dot(adj_matrix_x)
        
        s = R.copy()
        s[s < 2] = 0   
        s[s == 2] = 1
        I = np.ones((s.shape[1], 1))
        s = csr_matrix(s)
        I = csr_matrix(I)
        x = np.dot(s,I)
        x = x.todense()
        
        if (len(inc_matrix_co) == 0):
            meet_the_condition = False
            break

    inc_matrix_co = inc_matrix_co.transpose()
    if meet_the_condition:  
        print('')
        print("K-Truss matrix")
        print(inc_matrix_co)
        inc_matrix_co_sparse = csr_matrix(inc_matrix_co)
        adj_matrix_co_kTruss = inc_matrix_co_sparse.dot(inc_matrix_co_sparse.transpose())
        adj_matrix_co_kTruss = adj_matrix_co_kTruss.todense()

        for i in range(adj_matrix_co_kTruss.shape[0]):
            adj_matrix_co_kTruss[i,i] = 0

        #Visualizing K-Truss Graph
        rows, cols = np.where(adj_matrix_co_kTruss == 1)
        edges = zip(rows.tolist(), cols.tolist())
        gr = nx.Graph()
        gr.add_edges_from(edges)
        labels = nx.get_node_attributes(gr, 'label')
        nx.draw(gr, 
                node_size=30, 
                labels = labels, 
                with_labels = True)
        plt.show()
    else:
        print('')
        print("There is no subgraph meet the condition")
    k_value = k_value + 1

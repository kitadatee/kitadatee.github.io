#BEGIN OF IMPLEMENTING K-TRUSS
k_value = 3
print("K-truss with k value = " + str(k_value))
# Initializing E
inc_matrix = np.zeros((nb_of_edge, nb_of_vertex), dtype = int)
for i in range(dataset_inc.shape[0]):
    inc_matrix[dataset_inc.iat[i, 1] - 1, dataset_inc.iat[i, 0] - 1] = 1
inc_matrix_sparse = csr_matrix(inc_matrix)

# R = EA
R = inc_matrix_sparse.dot(adj_matrix)
# Replace the value of 2 by 1 and others by 0
s = R.copy()
s[s < 2] = 0
s[s == 2] = 1
# Initializing a matrix of 1
I = np.ones((s.shape[1], 1))
x = s.dot(I)
while (np.min(x) < k_value - 2):
    i_x = np.where(x >= k_value - 2)[0]
    i_xc = np.where(x < k_value - 2)[0]
    
    inc_matrix_x = np.delete(inc_matrix, i_x, axis = 0)
    inc_matrix = np.delete(inc_matrix, i_xc, axis = 0)
    adj_matrix_x = np.zeros((inc_matrix_x.shape[1], inc_matrix_x.shape[1]), dtype = int)
    for i in range(inc_matrix_x.shape[0]):
        position = np.asarray(np.nonzero(inc_matrix_x[i]))
        adj_matrix_x[position[0][0]][position[0][1]] = 1
        adj_matrix_x[position[0][1]][position[0][0]] = 1
    
    inc_matrix_sparse = csr_matrix(inc_matrix) 
    R = np.delete(R, i_xc, axis = 0) #Rxc   
    R = R - inc_matrix_sparse.dot(adj_matrix_x)

    s = R.copy()
    s[s < 2] = 0   
    s[s == 2] = 1

    I = np.ones((s.shape[1], 1))
    x = np.dot(s,I)
    if (len(inc_matrix) == 0):
        print("There is no subgraph meet the condition")
        break

inc_matrix = inc_matrix.transpose()
print(inc_matrix)

inc_matrix_sparse = csr_matrix(inc_matrix)
adj_matrix_kTruss = inc_matrix_sparse.dot(inc_matrix_sparse.transpose())
adj_matrix_kTruss = adj_matrix_kTruss.todense()

for i in range(adj_matrix_kTruss.shape[0]):
    adj_matrix_kTruss[i,i] = 0

#Visualizing K-Truss Graph
rows, cols = np.where(adj_matrix_kTruss == 1)
edges = zip(rows.tolist(), cols.tolist())
gr = nx.Graph()
gr.add_edges_from(edges)
labels = nx.get_node_attributes(gr, 'label')
nx.draw(gr, node_size=100, labels = labels, with_labels = True)
